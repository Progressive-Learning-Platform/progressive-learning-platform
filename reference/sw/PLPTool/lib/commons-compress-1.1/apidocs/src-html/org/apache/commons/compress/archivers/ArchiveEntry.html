019    package org.apache.commons.compress.archivers;
020    
021    import java.util.Date;
022    
023    /**
024     * Represents an entry of an archive.
025     */
026    public interface ArchiveEntry {
027    
028        /** The name of the entry in the archive. May refer to a file or directory or other item */
029        public String getName();
030    
031        /** The (uncompressed) size of the entry. May be -1 (SIZE_UNKNOWN) if the size is unknown */
032        public long getSize();
033    
034        /** Special value indicating that the size is unknown */
035        public static final long SIZE_UNKNOWN = -1;
036    
037        /** True if the entry refers to a directory */
038        public boolean isDirectory();
039    
040        /**
041         * The last modified date of the entry.
042         * 
043         * @since Apache Commons Compress 1.1
044         */
045        public Date getLastModifiedDate();
046    }
