019    package org.apache.commons.compress.archivers.jar;
020    
021    import java.io.IOException;
022    import java.io.OutputStream;
023    
024    import org.apache.commons.compress.archivers.ArchiveEntry;
025    import org.apache.commons.compress.archivers.zip.JarMarker;
026    import org.apache.commons.compress.archivers.zip.ZipArchiveEntry;
027    import org.apache.commons.compress.archivers.zip.ZipArchiveOutputStream;
028    
029    /**
030     * Subclass that adds a special extra field to the very first entry
031     * which allows the created archive to be used as an executable jar on
032     * Solaris.
033     * 
034     * @NotThreadSafe
035     */
036    public class JarArchiveOutputStream extends ZipArchiveOutputStream {
037    
038        private boolean jarMarkerAdded = false;
039    
040        public JarArchiveOutputStream(final OutputStream out) {
041            super(out);
042        }
043    
044        // @throws ClassCastException if entry is not an instance of ZipArchiveEntry
045        public void putArchiveEntry(ArchiveEntry ze) throws IOException {
046            if (!jarMarkerAdded) {
047                ((ZipArchiveEntry)ze).addAsFirstExtraField(JarMarker.getInstance());
048                jarMarkerAdded = true;
049            }
050            super.putArchiveEntry(ze);
051        }
052    }
