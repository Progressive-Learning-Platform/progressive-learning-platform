019    package org.apache.commons.compress.compressors;
020    
021    /**
022     * Compressor related exception
023     */
024    public class CompressorException extends Exception {
025    
026        /** Serial */
027        private static final long serialVersionUID = -2932901310255908814L;
028    
029        /**
030         * Constructs a new exception with the specified detail message. The cause
031         * is not initialized.
032         * 
033         * @param message
034         *            the detail message
035         */
036        public CompressorException(String message) {
037            super(message);
038        }
039    
040        /**
041         * Constructs a new exception with the specified detail message and cause.
042         * 
043         * @param message
044         *            the detail message
045         * @param cause
046         *            the cause
047         */
048        public CompressorException(String message, Throwable cause) {
049            super(message, cause);
050        }
051    }
